\usubsection{Conventions}

\usubsubsection{Signals Naming}

\begin{itemize}
    \item Input and output signals name end with "i" or "o" to indicate if they are input or output signals respectively.
    
    Example: \mintinline{vhdl}{clk_i} or \mintinline{vhdl}{m_axi_wdata_o}.


    \item When an explanation is applicable for both read and write channels, we used X.
    
    Example: \mintinline{vhdl}{m_axi_Xdata_o}
\end{itemize}



\usubsection{List of acronyms}
\begin{itemize}
    \item AW: Write Request
    \item W: Write
    \item B: Response
    \item AR: Read Request
    \item R: Read
    
\end{itemize}
